```csharp
private async void OnClearAgentSkillsClick(object sender, RoutedEventArgs e)
{
    if (sender is not Button btn || btn.Tag is not string agentKey) return;

    var agentNames = new Dictionary<string, string>
    {
        ["app"] = "App-Agent（应用匹配）",
        ["computer"] = "Comp-Agent（系统工具）",
        ["browser"] = "Browser-Agent（搜索/浏览器）",
        ["file"] = "File-Agent（文件操作）"
    };

    var name = agentNames.GetValueOrDefault(agentKey, agentKey);

    var dialog = new ContentDialog
    {
        Title = $"清除 {name} 的学习记录",
        Content = $"确定要删除「{name}」下的所有技能吗？此操作不可撤销。",
        PrimaryButtonText = "清除",
        CloseButtonText = "取消",
        DefaultButton = ContentDialogButton.Close,
        XamlRoot = XamlRoot
    };

    if (await dialog.ShowAsync() == ContentDialogResult.Primary)
    {
        var skills = App.SkillLibraryService.GetByAgentType(ParseAgentType(agentKey));
        foreach (var s in skills.ToList())
            App.SkillLibraryService.Delete(s.Id);
        PopulateSkillList();
    }
}

private static AgentType ParseAgentType(string key) => key.ToLowerInvariant() switch
{
    "app" => AgentType.App,
    "computer" => AgentType.Computer,
    "browser" => AgentType.Browser,
    "file" => AgentType.File,
    _ => AgentType.Computer
};
```
